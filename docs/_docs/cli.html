cli #

Usage example:

module main

import os
import cli

fn main() {
    mut app := cli.Command{
        name: 'example-app'
        description: 'example-app'
        execute: fn (cmd cli.Command) ? {
            println('hello app')
            return
        }
        commands: [
            cli.Command{
                name: 'sub'
                execute: fn (cmd cli.Command) ? {
                   println('hello subcommand')
                   return
                }
            },
        ]
    }
    app.setup()
    app.parse(os.args)
}

struct Command {
pub mut:
	name            string

	version         string

	sort_commands   bool
	parent          &Command = 0

}
